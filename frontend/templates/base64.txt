            question.save()

        print("updated question content for question_ids", question_ids)


def convert_file_details_to_dtos(
        file_details: Dict[str, str],
        parent_file_id: Optional[str]) -> List[FileDetailsDTO]:
    file_details_dtos = []
    file_id = str(uuid.uuid4())
    file_details_dto = FileDetailsDTO(
        id=file_id,
        file_name=file_details['file_name'],
        file_content=file_details['file_content'],
        file_type=file_details['file_type'],
        parent_file_id=parent_file_id,
    )
    file_details_dtos.append(file_details_dto)
    for child_file in file_details.get('child_files', []):
        child_file_details_dtos = convert_file_details_to_dtos(
            file_details=child_file, parent_file_id=file_id,
        )
        file_details_dtos += child_file_details_dtos
        
    return file_details_dtos
